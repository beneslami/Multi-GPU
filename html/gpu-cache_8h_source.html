146  void init(char * config, FuncCache status)
147  {
148  cache_status= status;
149  assert( config );
150  char rp, wp, ap, mshr_type, wap, sif;
151 
152  int ntok = sscanf(config,"%u:%u:%u,%c:%c:%c:%c:%c,%c:%u:%u,%u:%u,%u",
153  &m_nset, &m_line_sz, &m_assoc, &rp, &wp, &ap, &wap,
154  &sif, &mshr_type, &m_mshr_entries,&m_mshr_max_merge,
155  &m_miss_queue_size,&m_result_fifo_entries,
156  &m_data_port_width);
157 
158  if ( ntok < 11 ) {
159  if ( !strcmp(config,"none") ) {
160  m_disabled = true;
161  return;
162  }
163  exit_parse_error();
164  }
165  switch (rp) {
166  case 'L': m_replacement_policy = LRU; break;
167  case 'F': m_replacement_policy = FIFO; break;
168  default: exit_parse_error();
169  }
170  switch (wp) {
171  case 'R': m_write_policy = READ_ONLY; break;
172  case 'B': m_write_policy = WRITE_BACK; break;
173  case 'T': m_write_policy = WRITE_THROUGH; break;
174  case 'E': m_write_policy = WRITE_EVICT; break;
175  case 'L': m_write_policy = LOCAL_WB_GLOBAL_WT; break;
176  default: exit_parse_error();
177  }
178  switch (ap) {
179  case 'm': m_alloc_policy = ON_MISS; break;
180  case 'f': m_alloc_policy = ON_FILL; break;
181  default: exit_parse_error();
182  }
183  switch (mshr_type) {
184  case 'F': m_mshr_type = TEX_FIFO; assert(ntok==13); break;
185  case 'A': m_mshr_type = ASSOC; break;
186  default: exit_parse_error();
187  }
188  m_line_sz_log2 = LOGB2(m_line_sz);
189  m_nset_log2 = LOGB2(m_nset);
190  m_valid = true;
191 
192  switch(wap){
193  case 'W': m_write_alloc_policy = WRITE_ALLOCATE; break;
194  case 'N': m_write_alloc_policy = NO_WRITE_ALLOCATE; break;
195  default: exit_parse_error();
196  }
197 
198  // detect invalid configuration
199  if (m_alloc_policy == ON_FILL and m_write_policy == WRITE_BACK) {
200  // A writeback cache with allocate-on-fill policy will inevitably lead to deadlock:
201  // The deadlock happens when an incoming cache-fill evicts a dirty
202  // line, generating a writeback request. If the memory subsystem
203  // is congested, the interconnection network may not have
204  // sufficient buffer for the writeback request. This stalls the
205  // incoming cache-fill. The stall may propagate through the memory
206  // subsystem back to the output port of the same core, creating a
207  // deadlock where the wrtieback request and the incoming cache-fill
208  // are stalling each other.
209  assert(0 && "Invalid cache configuration: Writeback cache cannot allocate new line on fill. ");
210  }
211 
212  // default: port to data array width and granularity = line size
213  if (m_data_port_width == 0) {
214  m_data_port_width = m_line_sz;
215  }
216  assert(m_line_sz % m_data_port_width == 0);
217  switch(sif){
218  case 'H': m_set_index_function = FERMI_HASH_SET_FUNCTION; break;
219  case 'C': m_set_index_function = CUSTOM_SET_FUNCTION; break;
220  case 'L': m_set_index_function = LINEAR_SET_FUNCTION; break;
221  default: exit_parse_error();
222  }
223 
224  }
225  bool disabled() const { return m_disabled;}
226  unsigned get_line_sz() const
227  {
228  assert( m_valid );
229  return m_line_sz;
230  }

108 template<typename T>
109 void Channel<T>::WriteOutputs() {
110  _output = 0;
111  if(_wait_queue.empty()) {
112  return;
113  }
114  pair<int, T *> const & item = _wait_queue.front();
115  int const & time = item.first;
116  if(GetSimTime() < time) {
117  return;
118  }
119  assert(GetSimTime() == time);
120  _output = item.second;
121  assert(_output);
122  _wait_queue.pop();
123 }

113 struct type_info_key_compare {
114  bool operator()( const type_info_key &a, const type_info_key &b ) const
115  {
116  assert( a.m_init && b.m_init );
117  if ( a.m_space_spec < b.m_space_spec ) return true;
118  if ( a.m_scalar_type_spec < b.m_scalar_type_spec ) return true;
119  if ( a.m_vector_spec < b.m_vector_spec ) return true;
120  if ( a.m_alignment_spec < b.m_alignment_spec ) return true;
121  if ( a.m_extern_spec < b.m_extern_spec ) return true;
122  if ( a.m_array_dim < b.m_array_dim ) return true;
123  if ( a.m_is_function < b.m_is_function ) return true;
124 
125  return false;
126  }

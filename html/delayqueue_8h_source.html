44 template <class T>
45 class fifo_pipeline {
46 public:
47  fifo_pipeline(const char* nm, unsigned int minlen, unsigned int maxlen )
48  {
49  assert(maxlen);
50  m_name = nm;
51  m_min_len = minlen;
52  m_max_len = maxlen;
53  m_length = 0;
54  m_n_element = 0;
55  m_head = NULL;
56  m_tail = NULL;
57  for (unsigned i=0;i<m_min_len;i++)
58  push(NULL);
59  }
60 
61  ~fifo_pipeline()
62  {
63  while (m_head) {
64  m_tail = m_head;
65  m_head = m_head->m_next;
66  delete m_tail;
67  }
68  }
69 
70  void push(T* data )
71  {
72  assert(m_length < m_max_len);
73  if (m_head) {
74  if (m_tail->m_data || m_length < m_min_len) {
75  m_tail->m_next = new fifo_data<T>();
76  m_tail = m_tail->m_next;
77  m_length++;
78  m_n_element++;
79  }
80  } else {
81  m_head = m_tail = new fifo_data<T>();
82  m_length++;
83  m_n_element++;
84  }
85  m_tail->m_next = NULL;
86  m_tail->m_data = data;
87  }
88 
89  T* pop()
90  {
91  fifo_data<T>* next;
92  T* data;
93  if (m_head) {
94  next = m_head->m_next;
95  data = m_head->m_data;
96  if ( m_head == m_tail ) {
97  assert( next == NULL );
98  m_tail = NULL;
99  }
100  delete m_head;
101  m_head = next;
102  m_length--;
103  if (m_length == 0) {
104  assert( m_head == NULL );
105  m_tail = m_head;
106  }
107  m_n_element--;
108  if (m_min_len && m_length < m_min_len) {
109  push(NULL);
110  m_n_element--; // uncount NULL elements inserted to create delays
111  }
112  } else {
113  data = NULL;
114  }
115  return data;
116  }
117 
118  T* top() const
119  {
120  if (m_head) {
121  return m_head->m_data;
122  } else {
123  return NULL;
124  }
125  }

127  void set_min_length(unsigned int new_min_len)
128  {
129  if (new_min_len == m_min_len) return;
130 
131  if (new_min_len > m_min_len) {
132  m_min_len = new_min_len;
133  while (m_length < m_min_len) {
134  push(NULL);
135  m_n_element--; // uncount NULL elements inserted to create delays
136  }
137  } else {
138  // in this branch imply that the original min_len is larger then 0
139  // ie. head != 0
140  assert(m_head);
141  m_min_len = new_min_len;
142  while ((m_length > m_min_len) && (m_tail->m_data == 0)) {
143  fifo_data<T> *iter;
144  iter = m_head;
145  while (iter && (iter->m_next != m_tail))
146  iter = iter->m_next;
147  if (!iter) {
148  // there is only one node, and that node is empty
149  assert(m_head->m_data == 0);
150  pop();
151  } else {
152  // there are more than one node, and tail node is empty
153  assert(iter->m_next == m_tail);
154  delete m_tail;
155  m_tail = iter;
156  m_tail->m_next = 0;
157  m_length--;
158  }
159  }
160  }
161  }

178  }

52 class powerComponents

55  double dynamic;
56  double leakage;
57  double gate_leakage;
58  double short_circuit;
59  double longer_channel_leakage;

61  powerComponents() : dynamic(0), leakage(0), gate_leakage(0), short_circuit(0), longer_channel_leakage(0) { }
62  powerComponents(const powerComponents & obj) { *this = obj; }
63  powerComponents & operator=(const powerComponents & rhs)
64  {
65  dynamic = rhs.dynamic;
66  leakage = rhs.leakage;
67  gate_leakage = rhs.gate_leakage;
68  short_circuit = rhs.short_circuit;
69  longer_channel_leakage = rhs.longer_channel_leakage;
70  return *this;
71  }
72  void reset() { dynamic = 0; leakage = 0; gate_leakage = 0; short_circuit = 0;longer_channel_leakage = 0;}
73 
74  friend powerComponents operator+(const powerComponents & x, const powerComponents & y);
75  friend powerComponents operator*(const powerComponents & x, double const * const y);
76 };
